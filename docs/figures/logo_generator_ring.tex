\tikzset{
pics/ve_arrow/.style args={#1}{
code={
  \pgfmathsetmacro{\xx}{{cos(#1)}}
  \pgfmathsetmacro{\yy}{{sin(#1)}}
  \pgfmathsetmacro{\rad}{0.15}
  \pgfmathsetmacro{\xxt}{{(1+\rad)*cos(#1)}}
  \pgfmathsetmacro{\yyt}{{(1+\rad)*sin(#1)}}
  \draw [draw=black!1, line width = 1, opacity=1.0, line cap=round] (\xxt, \yyt) arc (#1:#1-180:\rad);
  \draw [draw=black!1, line width=1, opacity=1.0,] (\xx, \yy) arc (#1:#1+112:1);
  \draw [draw=black!1, fill=black!1, opacity=1.0,] (\xx, \yy) circle (0.45*\rad);
}
}
}
\begin{tikzpicture}
% Typeset in 12pt font for good size
\foreach \theta in {0, 120, 240}
{
  \draw pic{ve_arrow={\theta}};
}
